# ------------------------------------------------
# Choose compiler and set options 
# ------------------------------------------------
GCC = gcc
CFLAGS = -Wall -fno-builtin -I../utl -I../basop -I../src -std=c99 
LD_FLAGS =

# ------------------------------------
#  Lists of object files
# ------------------------------------
G722ENC_OBJ = \
funcg722.o \
g722.o \
../basop/basop32.o \
../basop/control.o \
../basop/count.o \
../basop/enh1632.o \
encg722.o

G722DEC_OBJ = \
funcg722.o \
g722.o \
softbit.o \
../basop/basop32.o \
../basop/control.o \
../basop/count.o \
../basop/enh1632.o \
decg722.o

G722DEMO_OBJ = \
funcg722.o \
g722.o \
softbit.o \
../basop/basop32.o \
../basop/control.o \
../basop/count.o \
../basop/enh1632.o \
g722demo.o

# ------------------------------------
# Targets 
# ------------------------------------
all: g722demo encg722 decg722

clean:
	@ rm -f `find .. -name '*~'` `find ..  -name '*.o'`

veryclean: clean
	$(RM) g722demo* encg722* decg722*

# ------------------------------------
# Dependencies and rules 
# ------------------------------------
g722demo: $(G722DEMO_OBJ)
	$(CC) $(CC_OPT) -o $@ $^ $(LD_OPT) -lm

encg722: $(G722ENC_OBJ)
	$(CC) $(CC_OPT) -o $@ $^ $(LD_OPT) -lm

decg722: $(G722DEC_OBJ)
	$(CC) $(CC_OPT) -o $@ $^ $(LD_OPT) -lm
